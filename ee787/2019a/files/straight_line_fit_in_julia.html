using LinearAlgebra
using PyPlot

n = 50
d = 2

theta_true = randn(d)
u = randn(n)
v = theta_true[1] .+ theta_true[2]*u + 0.2*randn(n)

figure(dpi=100)
plot(u, v, alpha=0.5, "o", label="Raw data")
xlabel("u")
ylabel("v")
legend()
grid()
show()
In [2]:
X = [ones(n) u]
y = v

theta_opt = X\y

y_hat = X*theta_opt
RMSE = norm(y-y_hat)/sqrt(n)
println("theta_true:", theta_true)
println("theta_opt:", theta_opt)
println("RMSE: ",RMSE)

theta_true:[-0.570999, -0.739793]
